🏗️ 技术架构

// 前端认证集成
src/app/ai-code-generator/
├── auth-integrated-ai-generator.js    # 认证集成的AI生成器
├── auth-integration-example.ts        # Angular集成示例
└── ai-code-generator.html             # 基础AI生成器

// 后端认证服务
backend/
├── routes/auth_routes.py              # 认证路由
├── utils/auth_middleware.py           # 认证中间件
├── models/user.py                     # 用户模型（含配额）
└── config/security.py                 # 安全配置

// 数据库表结构
users table:
- id (PK)
- username
- email
- password_hash
- role
- created_at
- last_login

ai_usage table:
- id (PK)
- user_id (FK)
- date
- usage_count
- created_at
